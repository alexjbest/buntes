\begin{equation*} \Gamma = \Gamma(2) = \{ A \in \PSL_2(\ZZ) : A = \id \pmod 2\} \end{equation*}
\begin{equation*} = \pi_1 (S' \smallsetminus \{0,1,\infty\}) \end{equation*}
